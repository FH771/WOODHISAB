rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // Global safety net match /{document=**} { allow read, write: if false; } // Auth helpers function isSignedIn() { return request.auth != null; } function isVerified() { return isSignedIn() && request.auth.token.email_verified == true; } function isOwner(userId) { return isSignedIn() && request.auth.uid == userId; } function isValidId(id) { return id is string && id.size() <= 128 && id.matches('^[a-zA-Z0-9_\\\\-]+$'); } function incoming() { return request.resource.data; } function existing() { return resource.data; } // Validation Blueprints function isValidWorker(data) { return data.keys().hasAll(['name', 'dailyRate', 'active', 'createdAt']) && data.name is string && data.name.size() > 0 && data.name.size() <= 100 && data.dailyRate is number && data.dailyRate >= 0 && data.active is bool && (!('phone' in data) || (data.phone is string && data.phone.size() <= 20)); } function isValidAttendance(data) { return data.keys().hasAll(['workerId', 'date', 'status', 'createdAt']) && data.workerId is string && data.date is string && data.date.matches('^\\\\d{4}-\\\\d{2}-\\\\d{2}$') && data.status in ['present', 'absent', 'half-day'] && (!('note' in data) || (data.note is string && data.note.size() <= 500)); } function isValidAdvance(data) { return data.keys().hasAll(['workerId', 'amount', 'date', 'createdAt']) && data.workerId is string && data.amount is number && data.amount > 0 && data.date is string && data.date.matches('^\\\\d{4}-\\\\d{2}-\\\\d{2}$') && (!('note' in data) || (data.note is string && data.note.size() <= 500)); } // Workers Collection match /workers/{workerId} { allow list: if isVerified(); allow get: if isVerified() && isValidId(workerId); allow create: if isVerified() && isValidId(workerId) && isValidWorker(incoming()) && incoming().createdAt == request.time; allow update: if isVerified() && isValidId(workerId) && ( // Identity Integrity: Workers info can be updated by verified users isValidWorker(incoming()) && incoming().createdAt == existing().createdAt && ( incoming().diff(existing()).affectedKeys().hasOnly(['name', 'dailyRate', 'phone', 'active']) || (incoming().diff(existing()).affectedKeys().hasOnly(['name', 'dailyRate', 'phone', 'active', 'updatedAt']) && incoming().updatedAt == request.time) ) ); allow delete: if isVerified() && isValidId(workerId); } // Attendance Collection match /attendance/{attendanceId} { allow list: if isVerified(); allow get: if isVerified() && isValidId(attendanceId); allow create: if isVerified() && isValidId(attendanceId) && isValidAttendance(incoming()) && incoming().createdAt == request.time; allow update: if isVerified() && isValidId(attendanceId) && isValidAttendance(incoming()) && ( incoming().createdAt == existing().createdAt && incoming().workerId == existing().workerId && incoming().date == existing().date && ( incoming().diff(existing()).affectedKeys().hasOnly(['status', 'note']) || (incoming().diff(existing()).affectedKeys().hasOnly(['status', 'note', 'updatedAt']) && incoming().updatedAt == request.time) ) ); allow delete: if isVerified() && isValidId(attendanceId); } // Advances Collection match /advances/{advanceId} { allow list: if isVerified(); allow get: if isVerified() && isValidId(advanceId); allow create: if isVerified() && isValidId(advanceId) && isValidAdvance(incoming()) && incoming().createdAt == request.time; allow update: if isVerified() && isValidId(advanceId) && isValidAdvance(incoming()) && ( incoming().createdAt == existing().createdAt && incoming().workerId == existing().workerId && ( incoming().diff(existing()).affectedKeys().hasOnly(['amount', 'date', 'note']) || (incoming().diff(existing()).affectedKeys().hasOnly(['amount', 'date', 'note', 'updatedAt']) && incoming().updatedAt == request.time) ) ); allow delete: if isVerified() && isValidId(advanceId); } } }
